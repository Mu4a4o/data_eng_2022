-- 1. Получите номера билетов,номера рейсов,моделм самолетов и города (ticket_flights,flights,aircrafts_data,airports_data)+

select
tf.ticket_no,
f.flight_no,
ad.model,
ad2.city
from bookings.flights as f
left join ticket_flights as tf
on f.flight_id =tf.flight_id
left join aircrafts_data as ad
on f.aircraft_code =ad.aircraft_code
left join airports_data as ad2
on f.arrival_airport =ad2.airport_code;

-- 2. Сгруппируйте по городам и классам обслуживания (join arrival_airport) и выясните их кол-во по городам за все время  (airports_data,flights,ticket_flights)+
--        Moscow Economy 200
--        Moscow Comfort 300
--        Moscow Business 50
--        ...

select ad.city, tf.fare_conditions,count(ad.city) co_
from bookings.ticket_flights as tf
left join flights as f
on tf.flight_id = f.flight_id
left join airports_data as ad
on f.arrival_airport =ad.airport_code
group by ad.city,tf.fare_conditions
order by ad.city, tf.fare_conditions;


-- 3. Выполните 2 п.п. только вместо  Economy -> Эконом, Comfort - Комфорт, Business - Бизнес (Lesson 5-func.sql)+

select ad.city,
CASE
	WHEN fare_conditions  = 'Economy' THEN 'Эконом'
	WHEN fare_conditions  = 'Comfort' THEN 'Комфорт'
	WHEN fare_conditions  = 'Business' THEN 'Бизнес'
	ELSE 'Без класса'
END Klass,
count(ad.city) co_
from bookings.ticket_flights as tf
left join flights as f
on tf.flight_id = f.flight_id
left join airports_data as ad
on f.arrival_airport =ad.airport_code
group by ad.city,tf.fare_conditions
order by ad.city, tf.fare_conditions;


-- 4. Выведите только пустые строки фактического временм вылетов И фактического времи прилётов (flights)+

select *
from bookings.flights
where actual_departure is null
or actual_arrival is null;


-- 5. Выведите только не пустые строки фактического временм вылетов И фактического времи прилётов (flights)+

select*
from bookings.flights
where actual_departure is not null
or actual_arrival is not null;


-- 6. Выведите только пустые строки фактического временм вылетов И фактического времи прилётов И вместо пустых строк сделайте текущее время (flights) (Lesson 10-other.sql)+

select *
,coalesce(actual_departure,now()) as actual_departure_not_null
,coalesce(actual_arrival,now()) as actual_arrival_not_null
FROM flights f;
